시라노소개팅 멤버쉽 14일 무료체험 // UTM 매개변수를 쿠키에 저장 function setUTMCookies() { var params = ['utm_source', 'utm_medium', 'utm_campaign', 'utm_term', 'utm_content']; var urlParams = new URLSearchParams(window.location.search); params.forEach(function(param) { if (urlParams.has(param)) { document.cookie = param + '=' + urlParams.get(param) + ';path=/;'; } }); } // 페이지 로드 시 실행 window.addEventListener('load', setUTMCookies);
